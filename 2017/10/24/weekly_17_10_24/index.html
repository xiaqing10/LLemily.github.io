《nodejs权威指南》学习收获

  1. nodejs通过非阻塞的IO机制，通过事件的回调函数来处理IO结果。当应用程序内部需要处理大量的并发的输入输出数据，程序内部不需要进行非常复杂的处理的时候，可以考虑使用nodejs

  2. 在安装node之后，可以在nodejs框架提供的交互式运行环境REPL中测试观察相关表达式的结果。

  3. Nodejs的全局作用域（global）和全局函数（例如setTimeout），全局变量(例如__dirname)

  4. 事件处理机制，nodejs中的event模块定义了一个EventEmitter类，在其定义的子类实例对象上，可以绑定事件处理函数，通过对事件的监听的响应来触发回调函数。

  5. 模块的使用，对于核心模块，例如http,net,dgram直接require使用；对于自定义的文件模块，需要用module.exports方式导出，在其他文件中require(路径名)使用，对于npm安装的第三方扩展模块，使用方式和核心模块相同。可以通过配置package.json的方法来进行安装

  6. 学习了Buffer类和文件系统，主要包括Buffer对象的创建以及和其他Javascript其他对象的转换。Nodejs中对文件及文件夹进行的读写，删除，修改等操作，以及文件流和管道的概念。

  • 使用net模块创建TCP通信
  • 使用dgram模块创建UDP通信
  • 使用http模块创建HTTP通信
  1. 使用express框架处理客户端提交的请求，通过客户端的请求数据来对mongodb数据库进行相关的操作

向数据库写入数据

服务器端：

var express = require('express');  //express框架引入和应用程序创建部分
var app = express();
var fs = require('fs');
var querystring = require('querystring');


var mongo = require('mongodb');   //数据库连接部分
var host = 'localhost';
var port = 27017;
var server = new mongo.Server(host,port,{auto_reconnect:true});
var db = new mongo.Db('node-mongo-examples',server,{safe:true});


app.get('/index.html',function(req,res){   //收到http://192.168.216.131:8888/index.html后发送html页面
    // res.writeHead(200,{'Content-Type':'text/html'});
    // res.write('<head><meta charset = "utf-8"/><title>使用post方法向服务器端提交数据</title></head>')
    // var file = fs.createReadStream('index.html');
    // file.pipe(res);
    res.sendFile(__dirname+'/index.html');
})


app.post('/index.html',function(req,res){    //收到post请求的数据后写入数据库
    req.on('data',function(data){
        console.log(data);
        var obj = querystring.parse(data.toString());
        db.open(function(err,db){
            if(err) throw err;
            else{
                console.log('成功连接到数据库');
                db.collection('names',function(err,collection){
                    if(err) throw err;
                    else{
                        collection.insert(obj,function(err,obj){
                            if(err) throw err;
                            else{
                                console.log(obj);
                                db.close();
                                res.send('写入数据库成功\n')
                            }
                        })
                    }
                })
            }
        })
    })
})

app.listen(8888);   //监听端口号为8888的接口

向数据库查询数据

服务器端：

var express = require('express');
var app = express();
var fs = require('fs');
var querystring = require('querystring');

var MongoClient = require('mongodb').MongoClient;
var DB_CONN_STR = 'mongodb://localhost:27017/node-mongo-examples';   

app.get('/index.html',function(req,res){
    res.sendFile(__dirname+'/index.html');
})

app.put('/index.html',function(req,res){
    req.on('data',function(data){
        var obj = JSON.parse(data.toString());
        console.log(obj);
        MongoClient.connect(DB_CONN_STR, function(err, db) {
             console.log("连接成功！");
            db.collection('names',function(err,collection){
                if(err) throw err;
                else{
                    // collection.insert(obj,function(err,obj){
                    //     if(err) throw err;
                    //     else{
                    //         console.log(obj);
                    //         db.close();
                    //         res.send('写入数据库成功\n')
                    //     }
                    // })
                    collection.find({lastname:obj.lastname},{fields:{lastname:1,_id:0}}).toArray(function(err,docs){
                        if(err) throw err;
                        else{
                            //res.send('从数据库取数据\n');
                            res.send(docs);
                        }
                    })
                }
            })          
        });
    })
})

app.listen(8888);

客户端：

<!DOCTYPE html>
<html>
<head>
    <title>使用put方法向服务器端提交数据</title>
    <script type="text/javascript">
        function submitData() {
            var obj = {
                firstname:document.getElementById('firstname').value,
                lastname:document.getElementById('lastname').value
            };
            var xhr = new XMLHttpRequest();
            xhr.open('put','index.html','true');

            xhr.onload = function(e){
                if(this.status == 200){
                    document.getElementById('result').innerHTML = this.response;
                }
            }
            xhr.send(JSON.stringify(obj));
            console.log(JSON.stringify(obj));
        }
    </script>
</head>
<body>
    <form id="form1" >
        <input type="text" name="firstname" id="firstname" value="杨"><br>
        <input type="text" name="lastname" id="lastname" value="彩清"><br>
        <input type="button" value="提交" onclick="submitData()">
    </form>
    <div id="result"></div>
</body>
</html>
